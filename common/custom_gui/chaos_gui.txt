custom_button = {
	name = gifts_of_chaos_1_1
	tooltip = gifts_of_chaos_1_1_tt
	potential = {
		always = yes
	}
	frame = {
		number = 1
		trigger = {
			NOT = {
				has_country_flag = chaos_gift_1_1_active
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			has_country_flag = chaos_gift_1_1_active
		}
	}
	trigger = {
		if = {
			limit = {
				not = {
					has_country_flag = chaos_gift_1_1_active
				}
			}
			check_variable = {
				chaos_power = 0
			}
		}
	}
	effect = {
		if = {
			limit = {
				has_country_flag = chaos_gift_1_1_active
				}
			clr_country_flag = chaos_gift_1_1_active
			remove_country_modifier = chaos_test_1
		}
		else_if = {
			limit = {
				NOT = {
					has_country_flag = chaos_gift_1_1_active
				}
			}
			set_country_flag = chaos_gift_1_1_active
			add_country_modifier = {
				name = chaos_test_1
				duration = -1
			}
		}
	}
}
